import pandas as pd

df = pd.DataFrame(
    [
        ["forward", 5],
        ["down", 5],
        ["forward", 8],
        ["up", 3],
        ["down", 8],
        ["forward", 2],
    ],
    columns=["direction", "value"],
)

df.convert_dtypes()


f = df["value"].where(df["direction"] == "forward").sum()
print(f)

up = df["value"].where(df["direction"] == "up").sum()
down = df["value"].where(df["direction"] == "down").sum()
depth = down - up
print(depth)

print(depth * f)
#+end_src

: 10.0
